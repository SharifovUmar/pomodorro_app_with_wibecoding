    def minimize_to_tray(self):
        """Сворачивает окно в системный трей"""
        self.hide()
        # Отключаем звук для уведомления
        self.tray_icon.showMessage(
            "Pomodoro Timer",
            "Приложение свернуто в системный трей. Дважды кликните по иконке для восстановления.",
            QSystemTrayIcon.Information | QSystemTrayIcon.NoSound,
            2000
        )
